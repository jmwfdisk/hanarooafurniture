rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ========== 헬퍼 함수 ==========
    
    // 관리자 확인 함수
    // 주의: users 컬렉션의 읽기 규칙에서는 순환 참조를 방지하기 위해 직접 확인
    function isAdmin() {
      return request.auth != null && 
        request.auth.uid != null &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }
    
    // 임직원 확인 함수
    function isEmployee() {
      return request.auth != null && 
        request.auth.uid != null &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.userType == 'employee' ||
         isAdmin());
    }
    
    // 인증된 사용자 확인 함수
    function isAuthenticated() {
      return request.auth != null && request.auth.uid != null;
    }
    
    // 본인 확인 함수
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // 승인된 사용자 확인 함수
    function isApprovedUser() {
      return isAuthenticated() &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'approved';
    }

    // 작성·관리 권한 맵 (없으면 빈 맵)
    function userPerms() {
      return (isAuthenticated() && exists(/databases/$(database)/documents/users/$(request.auth.uid)))
        ? get(/databases/$(database)/documents/users/$(request.auth.uid)).data.get('permissions', {})
        : {};
    }

    // 특정 영역 작성·관리 권한 (관리자/전체 포함)
    function permFor(area) {
      return isAdmin() ||
        (isAuthenticated() && (userPerms().get('all', false) == true || userPerms().get(area, false) == true));
    }

    // 인사기록 데이터관리자: 관리자 또는 명시적 'hr' 권한 부여자만
    // ('전체(all)' 권한으로는 불가 — 민감 인사정보라 결재 권한과 동일 원칙)
    function hrManager() {
      return isAdmin() || (isAuthenticated() && userPerms().get('hr', false) == true);
    }

    // 급여명세 관리자: 인사기록 데이터관리자 + 명시적 'pay'(급여명세 업로드) 권한 부여자
    // ('전체(all)' 권한으로는 불가 — 명세서 열람은 여전히 본인만, 이 권한은 업로드·삭제 전용)
    function payManager() {
      return hrManager() || (isAuthenticated() && userPerms().get('pay', false) == true);
    }

    // ========== 사용자 정보 (users) ==========
    
    match /users/{userId} {
      // 읽기: 본인, 관리자, 또는 인사기록 데이터관리자(명시적 'hr' 권한만 — '전체' 권한 불가)
      // 순환 참조를 방지하기 위해 isAdmin() 함수를 사용하지 않고 직접 확인
      allow read: if request.auth != null &&
        request.auth.uid != null && (
          // 본인 정보 읽기
          request.auth.uid == userId ||
          // 관리자가 다른 사용자 정보 읽기 (회원관리 기능)
          (exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true) ||
          // 인사기록('hr')·급여명세 업로드('pay') 권한자가 대상 임직원 선택을 위해 회원 목록 읽기
          (exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
           (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.get('permissions', {}).get('hr', false) == true ||
            get(/databases/$(database)/documents/users/$(request.auth.uid)).data.get('permissions', {}).get('pay', false) == true))
        );
      
      // 생성: 본인만 회원가입 시 생성 가능
      allow create: if request.auth != null && 
        request.auth.uid != null && 
        request.auth.uid == userId &&
        // 회원가입 시 필수 필드 검증
        request.resource.data.keys().hasAll(['uid', 'username', 'name', 'email', 'status', 'createdAt']) &&
        // UID 일치 확인
        request.resource.data.uid == request.auth.uid &&
        // 초기 상태는 pending이어야 함
        request.resource.data.status == 'pending' &&
        // 회원가입 시 isAdmin은 false여야 함 (보안)
        request.resource.data.isAdmin == false &&
        // username은 admin 사용 불가
        request.resource.data.username != 'admin' &&
        // 필수 필드 타입 검증
        request.resource.data.uid is string &&
        request.resource.data.username is string &&
        request.resource.data.name is string &&
        request.resource.data.email is string &&
        request.resource.data.status is string &&
        // 선택적 필드 검증
        (!('phone' in request.resource.data) || request.resource.data.phone is string) &&
        (!('userType' in request.resource.data) || request.resource.data.userType is string);
      
      // 수정: 본인 또는 관리자만 가능
      // 주의: uid/username 보호는 affectedKeys() 방식으로 검사한다.
      //       (resource.data.uid 처럼 없는 필드를 직접 비교하면 문서에 해당 필드가
      //        누락된 회원에서 규칙 평가가 실패해 관리자 수정까지 거부되는 문제가 있음)
      allow update: if request.auth != null &&
        request.auth.uid != null &&
        // 공통: uid, username 필드는 어떤 경우에도 변경 불가
        !('uid' in request.resource.data.diff(resource.data).affectedKeys()) &&
        !('username' in request.resource.data.diff(resource.data).affectedKeys()) &&
        (
          // 본인 정보 수정 (마이페이지 '내 정보 수정'): 소속·직급/직책만 허용.
          // 이름·연락처는 관리자만 수정 가능하며, org/position 외의 키
          // (name/phone/userType/empGroup/permissions/isAdmin/status 등)는 hasOnly(...)로 전부 차단.
          (request.auth.uid == userId &&
           request.resource.data.diff(resource.data).affectedKeys()
             .hasOnly(['org', 'position'])) ||
          // 관리자가 다른 사용자 정보 수정 (회원 승인/거절/소속/유형 등)
          (exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true)
        );
      
      // 삭제: 관리자만 가능
      allow delete: if request.auth != null && 
        request.auth.uid != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }

    // ========== A/S 게시판 (asPosts) ==========
    // 구조: doc('posts') 단일 문서에 { posts: [...] } 형태로 저장
    
    match /asPosts/{document} {
      // 읽기: 승인된 회원 또는 임직원만 (비인증 공개 읽기 차단 — 신청서에 연락처 등 개인정보 포함)
      // 주의: 단일 문서 {posts:[...]} 구조라 글 단위 서버 권한은 불가 → 승인 회원/임직원은
      //       배열 전체를 읽을 수 있고, '본인 글만 열람' 제한은 클라이언트(viewContent)가 담당.
      //       (자기 등록만 한 미승인 계정·비로그인 방문자는 이 규칙에서 차단됨)
      allow read: if isApprovedUser() || isEmployee();

      // 생성: 인증된 사용자만 가능 (초기 문서 생성)
      allow create: if isAuthenticated() &&
        // 필수 필드 검증
        request.resource.data.keys().hasAny(['posts']) &&
        // posts는 배열이어야 함
        request.resource.data.posts is list;
      
      // 수정: 인증된 사용자만 가능 (게시글 추가/수정/삭제)
      // 주의: 클라이언트에서 전체 posts 배열을 교체하므로, 
      // 개별 게시글 권한 검증은 클라이언트에서 처리해야 함
      allow update: if isAuthenticated() &&
        // posts 필드는 배열이어야 함
        request.resource.data.posts is list;
      
      // 삭제: 관리자만 가능 (전체 문서 삭제)
      allow delete: if isAdmin();
    }

    // ========== 직원 게시판 (staffPosts) ==========
    // 구조: doc(board) 형태로 각 게시판별 문서에 { posts: [...] } 형태로 저장
    
    match /staffPosts/{board} {
      // 읽기: 임직원 또는 관리자 — 단, 'staff-report'(업무보고 본사)는 'report' 권한 보유자만
      allow read: if isEmployee() &&
        (board != 'staff-report' || permFor('report'));

      // 생성/수정: 임직원(게시판 일반) — 단, 'as-result'는 'A/S처리결과', 'staff-report'는 'report' 권한 보유자만
      // (공지사항 'notice'·인증서 'cert'는 클라이언트에서 userCan('notice')/userCan('cert')로 제어 — 서버는 전 임직원 update 허용:
      //  조회수 증가(viewPost)도 update라서 서버까지 막으면 일반 직원이 '읽기'만 해도 저장 거부 알림이 떠 UX가 깨짐)
      allow create: if isEmployee() &&
        board is string &&
        request.resource.data.keys().hasAny(['posts']) &&
        request.resource.data.posts is list &&
        (board != 'as-result' || permFor('asResult')) &&
        (board != 'staff-report' || permFor('report'));

      // 수정: 동일 규칙 (클라이언트가 전체 posts 배열 교체)
      allow update: if isEmployee() &&
        board is string &&
        request.resource.data.posts is list &&
        (board != 'as-result' || permFor('asResult')) &&
        (board != 'staff-report' || permFor('report'));
      
      // 삭제: 관리자만 가능 (게시판 전체 삭제)
      allow delete: if isAdmin();
    }

    // ========== 활동사진첩 (activityPhotos) ==========
    // 구조: 개별 문서로 저장, 각 문서는 { filename, originalName, url, uploadedBy, uploaderName, createdAt }
    
    match /activityPhotos/{photoId} {
      // 읽기: 인증된 사용자만
      allow read: if isAuthenticated();
      
      // 생성: 인증된 사용자만
      allow create: if isAuthenticated() &&
        // 필수 필드 검증
        request.resource.data.keys().hasAll(['filename', 'url', 'uploadedBy', 'createdAt']) &&
        // uploadedBy는 본인 UID 또는 username이어야 함
        (request.resource.data.uploadedBy == request.auth.uid ||
         request.resource.data.uploadedBy == request.auth.token.email) &&
        // 필드 타입 검증
        request.resource.data.filename is string &&
        request.resource.data.url is string &&
        request.resource.data.uploadedBy is string &&
        request.resource.data.createdAt is string &&
        // 선택적 필드 검증
        (!('originalName' in request.resource.data) || request.resource.data.originalName is string) &&
        (!('uploaderName' in request.resource.data) || request.resource.data.uploaderName is string);
      
      // 수정: 본인이 업로드한 사진만 (관리자는 모든 사진 수정 가능)
      allow update: if isAuthenticated() && (
        // 관리자는 모든 사진 수정 가능
        isAdmin() ||
        // 본인이 업로드한 사진만 수정 가능
        (resource.data.keys().hasAny(['uploadedBy']) &&
         (resource.data.uploadedBy == request.auth.uid ||
          resource.data.uploadedBy == request.auth.token.email))
      ) &&
        // 수정 시 uploadedBy는 변경 불가
        (!('uploadedBy' in request.resource.data.diff(resource.data).affectedKeys()) ||
         request.resource.data.uploadedBy == resource.data.uploadedBy);
      
      // 삭제: 본인이 업로드한 사진만 (관리자는 모든 사진 삭제 가능)
      allow delete: if isAuthenticated() && (
        // 관리자는 모든 사진 삭제 가능
        isAdmin() ||
        // 본인이 업로드한 사진만 삭제 가능
        (resource.data.keys().hasAny(['uploadedBy']) &&
         (resource.data.uploadedBy == request.auth.uid ||
          resource.data.uploadedBy == request.auth.token.email))
      );
    }

    // ========== 자재관리 (materials) ==========
    // 구조: 개별 문서로 저장, 각 문서는 { sku, name, spec, unit, stock, safety, location, vendor, note, registeredAt, updatedAt, updatedBy }
    
    match /materials/{materialId} {
      // 읽기: 임직원 전체 (권한 불필요)
      allow read: if isEmployee();

      // 생성: '자재관리' 권한 보유자(또는 전체/관리자)만 가능
      allow create: if permFor('materials') &&
        // 필수 필드 검증
        request.resource.data.keys().hasAll(['sku', 'updatedAt']) &&
        // 필드 타입 검증
        request.resource.data.sku is string &&
        // 선택적 필드 검증
        (!('name' in request.resource.data) || request.resource.data.name is string) &&
        (!('spec' in request.resource.data) || request.resource.data.spec is string) &&
        (!('unit' in request.resource.data) || request.resource.data.unit is string) &&
        (!('stock' in request.resource.data) || request.resource.data.stock is number) &&
        (!('safety' in request.resource.data) || request.resource.data.safety is number) &&
        (!('location' in request.resource.data) || request.resource.data.location is string) &&
        (!('vendor' in request.resource.data) || request.resource.data.vendor is string) &&
        (!('note' in request.resource.data) || request.resource.data.note is string) &&
        (!('registeredAt' in request.resource.data) || request.resource.data.registeredAt is timestamp) &&
        (!('updatedBy' in request.resource.data) || request.resource.data.updatedBy is string);
      
      // 수정: '자재관리' 권한 보유자(또는 전체/관리자)만 가능
      allow update: if permFor('materials') &&
        // sku는 변경 불가
        (!('sku' in request.resource.data.diff(resource.data).affectedKeys()) ||
         request.resource.data.sku == resource.data.sku) &&
        // registeredAt는 변경 불가 (등록일은 고정)
        (!('registeredAt' in request.resource.data.diff(resource.data).affectedKeys()) ||
         request.resource.data.registeredAt == resource.data.registeredAt) &&
        // 필드 타입 검증
        (!('name' in request.resource.data) || request.resource.data.name is string) &&
        (!('spec' in request.resource.data) || request.resource.data.spec is string) &&
        (!('unit' in request.resource.data) || request.resource.data.unit is string) &&
        (!('stock' in request.resource.data) || request.resource.data.stock is number) &&
        (!('safety' in request.resource.data) || request.resource.data.safety is number) &&
        (!('location' in request.resource.data) || request.resource.data.location is string) &&
        (!('vendor' in request.resource.data) || request.resource.data.vendor is string) &&
        (!('note' in request.resource.data) || request.resource.data.note is string);
      
      // 삭제: '자재관리' 권한 보유자(또는 전체/관리자)만 가능
      allow delete: if permFor('materials');
    }

    // ========== 납품 일정관리 (deliverySchedule) ==========
    // 구조: doc('main') 단일 문서에 { notice, rows:[...], images:[...] } 형태로 저장
    match /deliverySchedule/{docId} {
      // 읽기: 임직원, 쓰기: 관리자 또는 '일정관리' 작성 권한 보유자
      allow read: if isEmployee();
      allow write: if permFor('schedule');
    }

    // ========== 앱 설정/메타 (appConfig) ==========
    // 예: doc('memberOrder')     = { order: [uid...] } 승인완료 회원 표시 순서
    //     doc('asAssignees')     = { list: [이름...] } A/S 출동담당자 목록
    //     doc('leaveApprovals')  = { months: { 'YYYY-MM': { hq:{title,name,date}, factory:{…} } } }
    //                              연월차 월별 결재 도장(본사=hq, 공장=factory; 구버전 flat 도장=본사로 해석)
    match /appConfig/{docId} {
      // 읽기: 임직원
      allow read: if isEmployee();
      // 쓰기: 기본은 관리자만. 단,
      //   - asAssignees(A/S 출동담당자 목록)  : 'asAssignee' 권한 보유자(또는 전체/관리자)도 수정 가능
      //   - leaveApprovals(연월차 월별 결재)  : '연월차 결재(본사/공장)' 권한 보유자(또는 전체/관리자)도 수정 가능
      allow write: if docId == 'asAssignees' ? permFor('asAssignee')
                   : (docId == 'leaveApprovals'
                       ? (permFor('leaveApprove') || permFor('leaveApproveFactory'))
                       : isAdmin());
    }

    // ========== 회사운영 달력 (companyCalendar) ==========
    // 구조: 개별 문서 { date:'YYYY-MM-DD', time, title, memo, createdBy, createdAt, periodStart?, periodEnd? }
    match /companyCalendar/{eventId} {
      // 읽기: 로그인 사용자
      allow read: if isAuthenticated();
      // 쓰기: 관리자 또는 '회사운영' 작성 권한 보유자
      allow write: if permFor('company');
    }

    // ========== 연월차일정 (leaveSchedule) ==========
    // 직원게시판 '연월차일정' 캘린더. 구조는 companyCalendar와 동일.
    // 추가·읽기 = 임직원 전체, 수정·삭제 = '연월차일정' 권한 보유자(또는 전체/관리자)
    //             + '연월차 결재(본사 leaveApprove/공장 leaveApproveFactory)' 권한 보유자(결재 완료된 달의 정정용)
    // ※ '결재 완료된 달은 결재자만 추가·수정·삭제' 월 잠금은 클라이언트에서 검사
    //   (appConfig/leaveApprovals 도장 기준 — 서버 규칙으로는 월 단위 잠금 미적용)
    match /leaveSchedule/{eventId} {
      allow read: if isEmployee();
      allow create: if isEmployee();
      allow update, delete: if permFor('leaveSchedule') || permFor('leaveApprove') || permFor('leaveApproveFactory');
    }

    // ========== 재고현황 (inventory) ==========
    // 구조: doc('main') = { groups: [{ name, rows:[{product,qty,stock,note}] }] }
    match /inventory/{docId} {
      // 읽기: 임직원, 쓰기: 관리자 또는 '재고현황' 작성 권한 보유자
      allow read: if isEmployee();
      allow write: if permFor('inventory');
    }

    // ========== 인사기록 (hrRecords) ==========
    // 구조: hrRecords/{uid} = { name, position, org, joinDate, phone, email, address,
    //        history:[{date,type,desc}], updatedAt, updatedBy }
    // 열람 = 본인 또는 인사기록 데이터관리자(관리자/명시적 'hr' 권한 — '전체' 권한 불가)
    // 작성·수정·삭제 = 인사기록 데이터관리자만 (일반 임직원은 본인 것도 수정 불가)
    match /hrRecords/{userId} {
      allow read: if isOwner(userId) || hrManager();
      allow write: if hrManager();
    }

    // ========== 급여지급명세서 (paySlips) ==========
    // 구조: paySlips/{uid} = { months: { 'YYYY-MM': { url, payDate, name, position, uploadedAt, uploadedBy } } }
    // 열람 = 본인만 (데이터관리자·관리자 포함 타인 열람 불가 — 쓰기 전용 업로드)
    // 작성·갱신 = 급여명세 관리자 payManager() = 관리자 / 명시적 'hr' / 명시적 'pay'(급여명세 업로드)
    match /paySlips/{userId} {
      allow read: if isOwner(userId);
      allow write: if payManager();
    }

    // 급여명세 인덱스: 어떤 귀속월·회사 명세서가 등록돼 있는지 여부만(URL·금액 등 내용 없음).
    // 급여명세 관리자가 삭제 전 존재 확인용으로 읽음 — 본문(paySlips)은 여전히 본인만 읽기 가능.
    match /paySlipsIndex/{userId} {
      allow read: if isOwner(userId) || payManager();
      allow write: if payManager();
    }

    // ========== 기타 컬렉션 ==========
    // 위에 명시되지 않은 모든 컬렉션은 접근 불가
    match /{document=**} {
      allow read, write: if false;
    }
  }
