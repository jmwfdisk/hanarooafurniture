rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // A/S 게시판 - 모든 사용자 접근 가능
    match /asPosts/{document} {
      allow read, write: if true;
    }

    // 나머지 모든 문서 - 로그인한 사용자만
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
