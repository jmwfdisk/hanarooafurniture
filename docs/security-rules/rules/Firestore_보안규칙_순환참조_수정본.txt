rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // 헬퍼 함수: 관리자 확인 (순환 참조 방지를 위해 users 규칙에서는 사용하지 않음)
    // 다른 컬렉션에서만 사용
    function isAdmin() {
      return request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }
    
    // 헬퍼 함수: 임직원 확인
    function isEmployee() {
      return request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.userType == 'employee' ||
         isAdmin());
    }
    
    // 사용자 정보 - 순환 참조 방지를 위해 isAdmin() 사용하지 않음
    match /users/{userId} {
      // 읽기: 본인 정보만 읽기 가능 (순환 참조 방지)
      allow read: if request.auth != null && request.auth.uid == userId;
      
      // 생성: 본인만 회원가입 시 생성 가능
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // 수정: 본인만 수정 가능 (순환 참조 방지)
      allow update: if request.auth != null && request.auth.uid == userId;
      
      // 삭제: 본인만 삭제 가능 (관리자 삭제는 별도 처리 필요)
      allow delete: if request.auth != null && request.auth.uid == userId;
    }

    // A/S 게시판 - 공개 읽기, 인증된 사용자만 작성
    match /asPosts/{document} {
      // 읽기: 모든 사용자 가능 (공개 게시판)
      allow read: if true;
      
      // 생성: 인증된 사용자만 가능
      allow create: if request.auth != null;
      
      // 수정/삭제: 본인이 작성한 게시글만 가능 (관리자는 모든 게시글 수정/삭제 가능)
      allow update, delete: if request.auth != null && (
        (!exists(resource.data.authorId) || resource.data.authorId == request.auth.uid) ||
        isAdmin()
      );
    }

    // 직원 게시판 - 임직원만 접근
    match /staffPosts/{board} {
      // 읽기/쓰기: 임직원 또는 관리자만 가능
      allow read, write: if isEmployee();
    }

    // 활동사진첩 - 인증된 사용자만 접근
    match /activityPhotos/{photoId} {
      // 읽기: 인증된 사용자만
      allow read: if request.auth != null;
      
      // 생성: 인증된 사용자만
      allow create: if request.auth != null;
      
      // 수정/삭제: 본인이 업로드한 사진만 (관리자는 모든 사진 수정/삭제 가능)
      allow update, delete: if request.auth != null && (
        (!exists(resource.data.uploadedBy) || resource.data.uploadedBy == request.auth.uid) ||
        isAdmin()
      );
    }
  }
}
