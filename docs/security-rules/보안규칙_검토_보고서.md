### 1. **Firestore 보안 규칙 - 사용자 정보 수정/삭제 권한 과다**

**현재 규칙 (위험):**
```javascript
match /users/{userId} {
  allow read: if request.auth != null;  // 모든 로그인 사용자가 모든 사용자 정보 읽기 가능
  allow create: if request.auth != null && request.auth.uid == userId;
  allow update, delete: if request.auth != null;  // ⚠️ 모든 로그인 사용자가 모든 사용자 정보 수정/삭제 가능!
}
```

**문제점:**
- ❌ **임의의 로그인 사용자가 다른 사용자의 정보를 수정/삭제할 수 있음**
- ❌ **모든 사용자 정보가 모든 로그인 사용자에게 노출됨**
- ❌ **관리자 권한 확인 없이 일반 사용자도 다른 사용자 정보 수정 가능**

**권장 수정:**
```javascript
match /users/{userId} {
  // 읽기: 본인 정보만 읽기 가능 (관리자는 예외)
  allow read: if request.auth != null && (
    request.auth.uid == userId || 
    get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true
  );
  
  // 생성: 본인만 회원가입 시 생성 가능
  allow create: if request.auth != null && request.auth.uid == userId;
  
  // 수정: 본인만 수정 가능 (관리자는 예외)
  allow update: if request.auth != null && (
    request.auth.uid == userId || 
    get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true
  );
  
  // 삭제: 관리자만 가능
  allow delete: if request.auth != null && 
    get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
}
```

### 2. **A/S 게시판 보안 규칙 - 무제한 쓰기 권한**

**현재 규칙:**
```javascript
match /asPosts/{document} {
  allow read, write: if true;  // ⚠️ 인증 없이 모든 사용자가 읽기/쓰기 가능
}
```

**문제점:**
- ❌ **익명 사용자도 게시글 작성/수정/삭제 가능**
- ❌ **스팸, 악성 게시글 작성 가능**
- ❌ **다른 사용자의 게시글 수정/삭제 가능**

**권장 수정:**
```javascript
match /asPosts/{document} {
  // 읽기: 모든 사용자 가능 (공개 게시판)
  allow read: if true;
  
  // 생성: 인증된 사용자만 가능
  allow create: if request.auth != null;
  
  // 수정/삭제: 본인이 작성한 게시글만 가능 (관리자는 예외)
  allow update, delete: if request.auth != null && (
    resource.data.authorId == request.auth.uid ||
    get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true
  );
}
```

**현재 규칙:**
```javascript
match /staffPosts/{board} {
  allow read, write: if request.auth != null;  // ⚠️ 모든 로그인 사용자가 읽기/쓰기 가능

```

**문제점:**
- ❌ **일반 사용자도 직원 게시판에 접근 가능**

**권장 수정:**
```javascript
match /staffPosts/{board} {

  allow read, write: if request.auth != null && (
    get(/databases/$(database)/documents/users/$(request.auth.uid)).data.userType == 'employee' ||
    get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true
  );
}
```

## 🟡 중간 수준 보안 취약점 (개선 권장)

### 4. **비밀번호 정책 약함**

**현재 검증:**
```javascript
if (password.length < 6) {
    alert('비밀번호는 6자 이상이어야 합니다.');
    return;
}
```

**문제점:**
- ⚠️ **최소 6자만 요구 (너무 약함)**
- ⚠️ **복잡도 요구사항 없음 (대문자, 소문자, 숫자, 특수문자)**
- ⚠️ **일반적인 비밀번호(123456, password 등) 허용**

**권장 개선:**
```javascript
// 비밀번호 강도 검증 함수 추가
function validatePassword(password) {
    if (password.length < 8) {
        return { valid: false, message: '비밀번호는 8자 이상이어야 합니다.' };
    }
    if (!/[A-Z]/.test(password)) {
        return { valid: false, message: '비밀번호에 대문자가 포함되어야 합니다.' };
    }
    if (!/[a-z]/.test(password)) {
        return { valid: false, message: '비밀번호에 소문자가 포함되어야 합니다.' };
    }
    if (!/[0-9]/.test(password)) {
        return { valid: false, message: '비밀번호에 숫자가 포함되어야 합니다.' };
    }
    if (!/[!@#$%^&*(),.?":{}|<>]/.test(password)) {
        return { valid: false, message: '비밀번호에 특수문자가 포함되어야 합니다.' };
    }
    // 일반적인 약한 비밀번호 차단
    const weakPasswords = ['password', '12345678', 'qwerty', 'abc123'];
    if (weakPasswords.includes(password.toLowerCase())) {
        return { valid: false, message: '너무 약한 비밀번호입니다.' };
    }
    return { valid: true };
}
```

### 5. **사용자 입력 검증 부족**

**현재 검증:**
```javascript
if (!username || !password || !name || !email) {
    alert('필수 항목을 모두 입력해주세요.');
    return;
}
if (username.length < 4) {
    alert('아이디는 4자 이상이어야 합니다.');
    return;
}
```

**문제점:**
- ⚠️ **이메일 형식 검증이 단순함** (`@` 포함 여부만 확인)
- ⚠️ **사용자명에 특수문자 제한 없음**
- ⚠️ **XSS 공격 방지 검증 없음**
- ⚠️ **전화번호 형식 검증 없음**

**권장 개선:**
```javascript
// 이메일 형식 검증
function validateEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
}

// 사용자명 검증 (영문, 숫자, 언더스코어만 허용)
function validateUsername(username) {
    const usernameRegex = /^[a-zA-Z0-9_]{4,20}$/;
    return usernameRegex.test(username);
}

// XSS 방지 (HTML 태그 제거)
function sanitizeInput(input) {
    const div = document.createElement('div');
    div.textContent = input;
    return div.innerHTML;
}

// 전화번호 형식 검증
function validatePhone(phone) {
    const phoneRegex = /^[0-9]{2,3}-?[0-9]{3,4}-?[0-9]{4}$/;
    return phoneRegex.test(phone);
}
```

### 6. **세션 관리 보안**

**현재 구현:**
```javascript
sessionStorage.setItem("loggedInUser", JSON.stringify(fullUserData));
sessionStorage.setItem("loggedIn", "true");
```

**문제점:**
- ⚠️ **세션 스토리지에 민감한 정보 저장** (이메일, 전화번호 등)
- ⚠️ **자동 로그아웃 시간이 15분으로 고정** (설정 불가)
- ⚠️ **세션 하이재킹 방지 메커니즘 없음**

**권장 개선:**
```javascript
// 민감한 정보는 세션 스토리지에 저장하지 않음
sessionStorage.setItem("loggedInUser", JSON.stringify({
    uid: user.uid,
    username: userData.username,
    name: userData.name,
    userType: userData.userType,
    isAdmin: userData.isAdmin
    // 이메일, 전화번호 등은 제외
}));

// 세션 토큰 검증 추가
function verifySession() {
    const lastActivity = sessionStorage.getItem("lastActivity");
    const now = Date.now();
    if (lastActivity && (now - parseInt(lastActivity)) > AUTO_LOGOUT_TIME) {
        logout();
        return false;
    }
    sessionStorage.setItem("lastActivity", now.toString());
    return true;
}
```

### 7. **에러 메시지 정보 노출**

**현재 구현:**
```javascript
if (error.code === 'auth/user-not-found') {
    alert('존재하지 않는 계정입니다.\n회원가입이 필요합니다.');
} else if (error.code === 'auth/wrong-password') {
    alert('비밀번호가 일치하지 않습니다.');
}
```

**문제점:**
- ⚠️ **에러 메시지로 계정 존재 여부를 확인할 수 있음** (사용자 열거 공격)
- ⚠️ **상세한 에러 정보가 공격자에게 유용할 수 있음**

**권장 개선:**
```javascript
// 일반적인 에러 메시지로 통일
if (error.code === 'auth/user-not-found' || 
    error.code === 'auth/wrong-password' || 
    error.code === 'auth/invalid-credential') {
    alert('아이디 또는 비밀번호가 올바르지 않습니다.');
} else {
    alert('로그인 중 오류가 발생했습니다.\n잠시 후 다시 시도해주세요.');
}
```

## 📝 수정된 보안 규칙 (전체)

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // 헬퍼 함수: 관리자 확인
    function isAdmin() {
      return request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }
    
    // 헬퍼 함수: 임직원 확인
    function isEmployee() {
      return request.auth != null && (
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.userType == 'employee' ||
        isAdmin()
      );
    }
    
    // A/S 게시판 - 공개 읽기, 인증된 사용자만 작성
    match /asPosts/{document} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && (
        resource.data.authorId == request.auth.uid || isAdmin()
      );
    }

    // 직원 게시판 - 임직원만 접근
    match /staffPosts/{board} {
      allow read, write: if isEmployee();
    }

    // 사용자 정보 - 본인만 읽기/수정, 관리자는 모든 권한
    match /users/{userId} {
      allow read: if request.auth != null && (
        request.auth.uid == userId || isAdmin()
      );
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && (
        request.auth.uid == userId || isAdmin()
      );
      allow delete: if isAdmin();
    }

    // 활동사진첩 - 인증된 사용자만 접근
    match /activityPhotos/{photoId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && (
        resource.data.uploadedBy == request.auth.uid || isAdmin()
      );
    }
  }
}
```
